代码注释

代码注释

  • 每个函数或方法都必须有严格的注释
    反模式示例:
      function doSomething(arg1,arg2,...){
      ...
  };
    最佳实践:
      /**
   * @desc 做某件事
   * @param {String} arg1 参数1
   * @param {Number} arg2 参数2
   * @param ...           参数n
   * @returns {undefined} 无返回值(没有指定返回值时返回undefined)
   */
     function doSomething(arg1,arg2,...){
        ...
    };
  • 构造器内声明的属性必须有清晰的注释
    反模式示例:
      /**
   * @desc 人类
   * @param {Object{name,age}} opts 参数
   */
    function Person(opts) {
        let config = {
            name: '无名',
            age: 20
        };
        this.config = Object.assign(config,opts);
        if(!this instanceof Person) {
            return new Person(opts);
        }
    }
    最佳实践:
      /**
   * @desc 人类
   * @param {Object{name,age}} opts 参数
   */
  function Person(opts) {
        let config = {
            name: '无名', //名字
            age: 20       //年龄 
        };
        this.config = Object.assign(config,opts);
        if(!this instanceof Person) {
            return new Person(opts);
        }
    }
